#include <rans.hpp>
#include <aslkdjfas>
#include <cassert>
int main() {
  std::string regex;
  std::cin >> regex;
  RANS rans(regex);
  if(!rans.ok()) {
    std::cout << rans.error() << std::endl;
    exit(0);
  }
  RANS::Value value; // avoid copy 
  std::text;
  while (std::cin >> text) {
    if (r.is_acceptable(text)) {
      std::cout << "val: " << r(text, value) << std::endl;
      std::cout << "rep: " << r(value, text) << std::endl;
    } else {
      std::cout << "text is not acceptable." << std::endl;
    }
  }
  return 0;
}
